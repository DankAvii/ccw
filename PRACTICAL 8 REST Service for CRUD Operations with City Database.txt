PRACTICAL 8: REST Service for CRUD Operations with City Database
AIM

To create a RESTful Web Service that performs CRUD operations (Create, Read, Update, Delete) on a City database with the fields:

City_Name
District_Name
Population

STEP 1: Create Web Application Project

Where: NetBeans IDE

Open NetBeans

Click

File → New Project

Select

Java Web → Web Application

Enter

Project Name : CityRESTService
Server : GlassFish Server

Click Finish

STEP 2: Create Package

Where: NetBeans IDE

Inside project create package

city

Location

Source Packages → New → Java Package
STEP 3: Create City Model Class

Where: NetBeans IDE

Create Java Class

City.java
Simplified Code
package city;

public class City {

    private int id;
    private String cityName;
    private String districtName;
    private int population;

    public City(){}

    public City(int id,String cityName,String districtName,int population){
        this.id=id;
        this.cityName=cityName;
        this.districtName=districtName;
        this.population=population;
    }

    public int getId(){
        return id;
    }

    public String getCityName(){
        return cityName;
    }

    public String getDistrictName(){
        return districtName;
    }

    public int getPopulation(){
        return population;
    }

}

Simplification:

Removed setters to make the code shorter.

STEP 5: Create REST Service Class

Where: NetBeans IDE

Create Java Class

CityService.java
Simplified Code
package city;

import javax.ws.rs.*;
import javax.ws.rs.core.MediaType;
import java.sql.*;

@Path("/cities")

public class CityService {

@GET
@Produces(MediaType.TEXT_PLAIN)

public String getCities(){

String result="";

try{

Connection con=DBConnection.getConnection();

Statement st=con.createStatement();

ResultSet rs=st.executeQuery("select * from city");

while(rs.next()){

result+=rs.getInt(1)+" "
+rs.getString(2)+" "
+rs.getString(3)+" "
+rs.getInt(4)+"\n";

}

}

catch(Exception e){
e.printStackTrace();
}

return result;

}

@POST
@Path("/add")

public String addCity(@QueryParam("name") String name,
@QueryParam("district") String district,
@QueryParam("population") int population){

try{

Connection con=DBConnection.getConnection();

PreparedStatement ps=con.prepareStatement(
"insert into city(city_name,district_name,population) values(?,?,?)");

ps.setString(1,name);
ps.setString(2,district);
ps.setInt(3,population);

ps.executeUpdate();

}

catch(Exception e){
e.printStackTrace();
}

return "City Added";

}

@PUT
@Path("/update")

public String updateCity(@QueryParam("id") int id,
@QueryParam("population") int population){

try{

Connection con=DBConnection.getConnection();

PreparedStatement ps=con.prepareStatement(
"update city set population=? where id=?");

ps.setInt(1,population);
ps.setInt(2,id);

ps.executeUpdate();

}

catch(Exception e){
e.printStackTrace();
}

return "City Updated";

}

@DELETE
@Path("/delete")

public String deleteCity(@QueryParam("id") int id){

try{

Connection con=DBConnection.getConnection();

PreparedStatement ps=con.prepareStatement(
"delete from city where id=?");

ps.setInt(1,id);

ps.executeUpdate();

}

catch(Exception e){
e.printStackTrace();
}

return "City Deleted";

}

}
